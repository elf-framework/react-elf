# react-elf
React component for elf-framework 

## Installation
```bash
npm install @react-elf/core
```


## Usage
```jsx
import React from 'react';
import { Button } from '@react-elf/button';

const App = () => {
  return (
    <Button>Click me</Button>
  );
};
```
